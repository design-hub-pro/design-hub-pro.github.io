# agent.py - NightHawk_v3
import torch
from neurobattle import BaseAgent, ACTIONS

class MyAgent(BaseAgent):
    def __init__(self):
        self.model = self.load_model("model_weights.pt")

    def decide_action(self, state: dict) -> str:
        # Process game state
        x = self.preprocess(state)

        # Get action probabilities from model
        with torch.no_grad():
            probs = self.model(x)

        # Return highest probability action
        return ACTIONS[probs.argmax().item()]
